program matmul{

vars{
	int i[2][2],b[2][2],c[2][2],m,n,p;
}

main{

	i[1][1] = 1;
	i[2][1] = 3;
	i[1][2] = 2;
	i[2][2] = 4;

	b[1][1] = 1;
	b[2][1] = 3;
	b[1][2] = 2;
	b[2][2] = 4;

	c[1][1] = 0;
	c[2][1] = 0;
	c[1][2] = 0;
	c[2][2] = 0;	

	for (m=1; m<=2 ; m=m+1;)
	{
		for (n=1; n<=2 ; n=n+1;)
		{
			for(p=1; p<=2 ; p=p+1;)
			{
				c[m][n] = c[m][n] + i[m][p] * b[p][n];
			}
		}
	}
	
}


}
