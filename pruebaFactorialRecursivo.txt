program factorialprograma {

    vars{
        int n, result;
    }

    module int factorial(int x) {
            {
                if(x==0) {
                    print("LLEGO A TOPE "&x);
                    return 1;
                } else {
                    print("Le falta "&x);
                    return x * factorial(x - 1);
                }
            }

        }


    main{
        print("factorial Solucion Recursivdo");
        print("Teclea un numero positivo:");
        input>>n;
        result = factorial(n);
        print(result);

    }

}
