program fibonacci {

vars{
int t1,t2,actual,n, next;
}



main{
print("Fibonnaci Solucion Iterativa");
print("Teclea un numero positivo:");
input>>n;

t1 = 0;
t2 = 1;
actual = 0;
next = 0;

while(next<=n)
{
    if (next == 1 or next == 0){
        print(next);
    } else {
        actual = t1+t2;
        print(actual);
        t1 = t2;
        t2 = actual;
    }
    next = next + 1;
}

}


}
